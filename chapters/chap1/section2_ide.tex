\LaTeX Workshop recipes 设置 latexmk:

\begin{lstlisting}
"latex-workshop.latex.tools": [
  {
    "name": "latexmk",
    "command": "latexmk",
    "args": [
      "--shell-escape",
      "-synctex=1",
      "-interaction=nonstopmode",
      "-file-line-error",
      "-xelatex",
      "%DOC%"
    ]
  },
],

  "latex-workshop.latex.recipes": [
  {
    "name": "latexmk",
    "tools": [
        "latexmk"
      ]
  },
],
\end{lstlisting}
